526int is_ipv4_address(const char* host)
527{
528 int dots = 0;
529 int digits = 0;
530 const char* ptr = host;
531
532 while (*ptr) {
533 if (*ptr == '.') {
534 if (digits == 0) return 0; // No digits before the dot
535 dots++;
536 digits = 0; // Reset digit count for the next block
537 }
538 else if (is_digit(*ptr)) {
539 digits++;
540 if (digits > 3) return 0; // Too many digits in one block
541 }
542 else {
543 return 0; // Invalid character for an IPv4 address
544 }
545 ptr++;
546 }
547
548 // There should be exactly 3 dots and at least one digit after the last dot
549 return (dots == 3 && digits > 0) ? 1 : 0;
550}

559int is_ipv6_address(const char* host)
560{
561 int colons = 0;
562 int double_colon = 0;
563 int hex_digits = 0;
564 const char* ptr = host;
565
566 while (*ptr) {
567 if (*ptr == ':') {
568 if (hex_digits == 0 && colons > 0 && *(ptr - 1) != ':') return 0; // No hex digits between colons
569 if (*(ptr + 1) == ':') {
570 if (double_colon) return 0; // Only one double colon allowed
571 double_colon = 1;
572 ptr++; // Skip the second colon of the double colon
573 }
574 colons++;
575 hex_digits = 0; // Reset hex digit count for the next block
576 }
577 else if (is_hex_digit(*ptr)) {
578 hex_digits++;
579 if (hex_digits > 4) return 0; // Each block in IPv6 should not exceed 4 hex digits
580 }
581 else {
582 return 0; // Invalid character for an IPv6 address
583 }
584 ptr++;
585 }
586
587 // Valid IPv6 addresses have between 2 and 7 colons, or a single double colon
588 return (colons >= 2 && colons <= 7) || (double_colon && colons < 7);
589}

672ULONG inet_addr(PCSTR cp)
673{
674 ULONG val, base, n;
675 UCHAR c;
676 ULONG parts[4], * pp = parts;
677 if (!cp) return INADDR_ANY;
678 if (!isdigit(*cp)) return INADDR_NONE;
679
680again:
681 /*
682 * Collect number up to ``.''.
683 * Values are specified as for C:
684 * 0x=hex, 0=octal, other=decimal.
685 */
686 val = 0;
687 base = 10;
688 if (*cp == '0')
689 {
690 if (*++cp == 'x' || *cp == 'X')
691 base = 16, cp++;
692 else
693 base = 8;
694 }
695 while ((c = *cp))
696 {
697 if (isdigit(c))
698 {
699 val = (val * base) + (c - '0');
700 cp++;
701 continue;
702 }
703 if (base == 16 && isxdigit(c))
704 {
705 val = (val << 4) + (c + 10 - (islower(c) ? 'a' : 'A'));
706 cp++;
707 continue;
708 }
709 break;
710 }
711 if (*cp == '.')
712 {
713 /*
714 * Internet format:
715 * a.b.c.d
716 * a.b.c (with c treated as 16-bits)
717 * a.b (with b treated as 24 bits)
718 */
719 if (pp >= parts + 4) return (INADDR_NONE);
720 *pp++ = val;
721 cp++;
722 goto again;
723 }
724 //Check for trailing characters.
725 if (*cp && !isspace((UCHAR)*cp)) return (INADDR_NONE);
726
727 *pp++ = val;
728 /*
729 * Concoct the address according to
730 * the number of parts specified.
731 */
732 n = (ULONG)(pp - parts);
733 switch (n)
734 {
735 case 1: /* a -- 32 bits */
736 val = parts[0];
737 break;
738
739 case 2: /* a.b -- 8.24 bits */
740 val = (parts[0] << 24) | (parts[1] & 0xffffff);
741 break;
742
743 case 3: /* a.b.c -- 8.8.16 bits */
744 val = (parts[0] << 24) | ((parts[1] & 0xff) << 16) | (parts[2] & 0xffff);
745 break;
746
747 case 4: /* a.b.c.d -- 8.8.8.8 bits */
748 val = (parts[0] << 24) | ((parts[1] & 0xff) << 16) | ((parts[2] & 0xff) << 8) | (parts[3] & 0xff);
749 break;
750
751 default:
752 return (INADDR_NONE);
753 }
754 val = htonl(val);
755 return (val);
756}

765ULONG convert_ip_to_htonl(PCSTR ip_address)
766{
767 ULONG parts[4] = { 0 }; // Array to store the four parts of the IP address
768 ULONG result = 0;
769 int part_index = 0;
770 int part_value = 0;
771
772 // Loop through each character in the string
773 while (*ip_address) {
774 if (isdigit(*ip_address)) {
775 // Build the current part by shifting the previous value by 10 and adding the new digit
776 part_value = part_value * 10 + (*ip_address - '0');
777 }
778 else if (*ip_address == '.') {
779 // Store the current part in the parts array
780 parts[part_index++] = part_value;
781 part_value = 0;
782 }
783 ip_address++;
784 }
785 // Store the last part
786 parts[part_index] = part_value;
787
788 // Combine the four parts into a single 32-bit integer
789 result = (parts[0] << 24) | (parts[1] << 16) | (parts[2] << 8) | parts[3];
790
791 // Convert to network byte order
792 return result;
793}

858 len = strlen(domain);
859 *qname++ = len;
860 memcpy(qname, domain, len);
861 qname += len;
862 *qname++ = 0x00;
863
864 // Type and Class
865 unsigned short qtype = htons(1); // Type A for IPv4 addresses
866 unsigned short qclass = htons(1); // Class IN
867 memcpy(qname, &qtype, 2);
868 memcpy(qname + 2, &qclass, 2);
869
870 *query_len = qname + 4 - query;
871
872 // Debug or logging purposes
873 if (subdomain_detected) {
874 printf("Subdomain detected.\n");
875 }
876 else {
877 printf("No subdomain detected.\n");
878 }
879
880 return query;
881}

891unsigned int parse_dns_response(unsigned char* response, int response_len)
892{
893 // Skip the DNS header and question section
894 unsigned char* answer_section = response + 12;
895 while (*answer_section != 0) answer_section++;
896 answer_section += 5; // Skip the null byte and QTYPE, QCLASS
897
898 // Parse the answer section
899 unsigned short answer_type;
900 unsigned short answer_data_len;
901
902 while (answer_section - response < response_len)
903 {
904 answer_section += 2; // Skip the NAME field
905 answer_type = ntohs(*(unsigned short*)(answer_section));
906 answer_section += 8; // Skip TYPE, CLASS, TTL
907 answer_data_len = ntohs(*(unsigned short*)(answer_section));
908 answer_section += 2;
909
910 if (answer_type == 1 && answer_data_len == 4) // A record
911 {
912 unsigned int ip_address = *(unsigned int*)(answer_section);
913 return ip_address;
914 }
915
916 answer_section += answer_data_len;
917 }
918
919 return 0; // No IP address found
920}
