538int is_ipv4_address(const char* host)
539{
540 int dots = 0;
541 int digits = 0;
542 const char* ptr = host;
543
544 while (*ptr) {
545 if (*ptr == '.') {
546 if (digits == 0) return 0; // No digits before the dot
547 dots++;
548 digits = 0; // Reset digit count for the next block
549 }
550 else if (is_digit(*ptr)) {
551 digits++;
552 if (digits > 3) return 0; // Too many digits in one block
553 }
554 else {
555 return 0; // Invalid character for an IPv4 address
556 }
557 ptr++;
558 }
559
560 // There should be exactly 3 dots and at least one digit after the last dot
561 return (dots == 3 && digits > 0) ? 1 : 0;
562}

571int is_ipv6_address(const char* host)
572{
573 int colons = 0;
574 int double_colon = 0;
575 int hex_digits = 0;
576 const char* ptr = host;
577
578 while (*ptr) {
579 if (*ptr == ':') {
580 if (hex_digits == 0 && colons > 0 && *(ptr - 1) != ':') return 0; // No hex digits between colons
581 if (*(ptr + 1) == ':') {
582 if (double_colon) return 0; // Only one double colon allowed
583 double_colon = 1;
584 ptr++; // Skip the second colon of the double colon
585 }
586 colons++;
587 hex_digits = 0; // Reset hex digit count for the next block
588 }
589 else if (is_hex_digit(*ptr)) {
590 hex_digits++;
591 if (hex_digits > 4) return 0; // Each block in IPv6 should not exceed 4 hex digits
592 }
593 else {
594 return 0; // Invalid character for an IPv6 address
595 }
596 ptr++;
597 }
598
599 // Valid IPv6 addresses have between 2 and 7 colons, or a single double colon
600 return (colons >= 2 && colons <= 7) || (double_colon && colons < 7);
601}

684ULONG inet_addr(PCSTR cp)
685{
686 ULONG val, base, n;
687 UCHAR c;
688 ULONG parts[4], * pp = parts;
689 if (!cp) return INADDR_ANY;
690 if (!isdigit(*cp)) return INADDR_NONE;
691
692again:
693 /*
694 * Collect number up to ``.''.
695 * Values are specified as for C:
696 * 0x=hex, 0=octal, other=decimal.
697 */
698 val = 0;
699 base = 10;
700 if (*cp == '0')
701 {
702 if (*++cp == 'x' || *cp == 'X')
703 base = 16, cp++;
704 else
705 base = 8;
706 }
707 while ((c = *cp))
708 {
709 if (isdigit(c))
710 {
711 val = (val * base) + (c - '0');
712 cp++;
713 continue;
714 }
715 if (base == 16 && isxdigit(c))
716 {
717 val = (val << 4) + (c + 10 - (islower(c) ? 'a' : 'A'));
718 cp++;
719 continue;
720 }
721 break;
722 }
723 if (*cp == '.')
724 {
725 /*
726 * Internet format:
727 * a.b.c.d
728 * a.b.c (with c treated as 16-bits)
729 * a.b (with b treated as 24 bits)
730 */
731 if (pp >= parts + 4) return (INADDR_NONE);
732 *pp++ = val;
733 cp++;
734 goto again;
735 }
736 //Check for trailing characters.
737 if (*cp && !isspace((UCHAR)*cp)) return (INADDR_NONE);
738
739 *pp++ = val;
740 /*
741 * Concoct the address according to
742 * the number of parts specified.
743 */
744 n = (ULONG)(pp - parts);
745 switch (n)
746 {
747 case 1: /* a -- 32 bits */
748 val = parts[0];
749 break;
750
751 case 2: /* a.b -- 8.24 bits */
752 val = (parts[0] << 24) | (parts[1] & 0xffffff);
753 break;
754
755 case 3: /* a.b.c -- 8.8.16 bits */
756 val = (parts[0] << 24) | ((parts[1] & 0xff) << 16) | (parts[2] & 0xffff);
757 break;
758
759 case 4: /* a.b.c.d -- 8.8.8.8 bits */
760 val = (parts[0] << 24) | ((parts[1] & 0xff) << 16) | ((parts[2] & 0xff) << 8) | (parts[3] & 0xff);
761 break;
762
763 default:
764 return (INADDR_NONE);
765 }
766 val = htonl(val);
767 return (val);
768}

777ULONG convert_ip_to_htonl(PCSTR ip_address)
778{
779 ULONG parts[4] = { 0 }; // Array to store the four parts of the IP address
780 ULONG result = 0;
781 int part_index = 0;
782 int part_value = 0;
783
784 // Loop through each character in the string
785 while (*ip_address) {
786 if (isdigit(*ip_address)) {
787 // Build the current part by shifting the previous value by 10 and adding the new digit
788 part_value = part_value * 10 + (*ip_address - '0');
789 }
790 else if (*ip_address == '.') {
791 // Store the current part in the parts array
792 parts[part_index++] = part_value;
793 part_value = 0;
794 }
795 ip_address++;
796 }
797 // Store the last part
798 parts[part_index] = part_value;
799
800 // Combine the four parts into a single 32-bit integer
801 result = (parts[0] << 24) | (parts[1] << 16) | (parts[2] << 8) | parts[3];
802
803 // Convert to network byte order
804 return result;
805}

852 memcpy(query + 4, &questions, 2);
853 memcpy(query + 6, &answer_rrs, 2);
854 memcpy(query + 8, &authority_rrs, 2);
855 memcpy(query + 10, &additional_rrs, 2);
856
857 // Build question
858 qname = query + 12;
859 const char* dot_pos = strchr(domain, '.');
860 int subdomain_detected = (dot_pos != NULL && strchr(dot_pos + 1, '.') != NULL);
861
862 while (dot_pos) {
863 len = dot_pos - domain;
864 *qname++ = len;
865 memcpy(qname, domain, len);
866 qname += len;
867 domain = dot_pos + 1;
868 dot_pos = strchr(domain, '.');
869 }
870 len = strlen(domain);
871 *qname++ = len;
872 memcpy(qname, domain, len);
873 qname += len;
874 *qname++ = 0x00;
875
876 // Type and Class
877 unsigned short qtype = htons(1); // Type A for IPv4 addresses
878 unsigned short qclass = htons(1); // Class IN
879 memcpy(qname, &qtype, 2);
880 memcpy(qname + 2, &qclass, 2);
881
882 *query_len = qname + 4 - query;
883
884 // Debug or logging purposes
885 if (subdomain_detected) {
886 printf("Subdomain detected.\n");
887 }
888 else {
889 printf("No subdomain detected.\n");
890 }
891
892 return query;
893}

903unsigned int parse_dns_response(unsigned char* response, int response_len)
904{
905 // Skip the DNS header and question section
906 unsigned char* answer_section = response + 12;
907 while (*answer_section != 0) answer_section++;
908 answer_section += 5; // Skip the null byte and QTYPE, QCLASS
909
910 // Parse the answer section
911 unsigned short answer_type;
912 unsigned short answer_data_len;
913
914 while (answer_section - response < response_len)
915 {
916 answer_section += 2; // Skip the NAME field
917 answer_type = ntohs(*(unsigned short*)(answer_section));
918 answer_section += 8; // Skip TYPE, CLASS, TTL
919 answer_data_len = ntohs(*(unsigned short*)(answer_section));
920 answer_section += 2;
921
922 if (answer_type == 1 && answer_data_len == 4) // A record
923 {
924 unsigned int ip_address = *(unsigned int*)(answer_section);
925 return ip_address;
926 }
927
928 answer_section += answer_data_len;
929 }
930
931 return 0; // No IP address found
932}
